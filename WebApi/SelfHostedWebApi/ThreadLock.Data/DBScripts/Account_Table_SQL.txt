CREATE TABLE ACCOUNT
(
	SEQ             NUMBER(10,0) NOT NULL,
	ACCOUNT_ID      VARCHAR2(64 BYTE) NOT NULL,
	BALANCE         NUMBER DEFAULT 0 NOT NULL,
	CREATED         DATE DEFAULT SYSDATE,
	CREATED_BY      VARCHAR2(80 BYTE),
	LAST_UPDATED    DATE DEFAULT SYSDATE,
	LAST_UPDATED_BY VARCHAR2(80 BYTE), 
	CONSTRAINT IDX_ACCOUNT_PK PRIMARY KEY ("SEQ")
);

-- Create/Recreate primary, unique and foreign key constraints 
alter table ACCOUNT add constraint IDX_ACCOUNT_UK unique (ACCOUNT_ID);

-- Create sequence 
CREATE SEQUENCE ACCOUNT_SEQ
       MINVALUE 1
       MAXVALUE 999999999999999999
       START WITH 1
       INCREMENT BY 1
	   CACHE 5
       CYCLE;
