CREATE TABLE ACCOUNT_HISTORY
(
	SEQ              NUMBER(18,0) NOT NULL,
	ACCOUNT_SEQ      NUMBER(10,0) NOT NULL,
	TRANSACTION_TYPE VARCHAR2(128 BYTE) NOT NULL,
	AMOUNT           NUMBER DEFAULT 0 NOT NULL,
	BALANCE          NUMBER DEFAULT 0 NOT NULL,
	ACCESSED         TIMESTAMP DEFAULT SYSDATE,
	ACCESSED_BY      VARCHAR2(80 BYTE)
);

-- Create sequence 
CREATE SEQUENCE ACCOUNT_HISTORY_SEQ
       MINVALUE 0
       MAXVALUE 999999999999999999
       START WITH 0
       INCREMENT BY 1
	   CACHE 5
       CYCLE;
