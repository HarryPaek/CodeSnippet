CREATE TABLE ACCOUNT_HISTORY
(
	SEQ              NUMBER(10,0) NOT NULL,
	TRANSACTION_TYPE VARCHAR2(128 BYTE) NOT NULL,
	AMOUNT           NUMBER(18,0) DEFAULT 0 NOT NULL,
	BALANCE          NUMBER(18,0) DEFAULT 0 NOT NULL,
	ACCESSED         DATE DEFAULT SYSDATE,
	ACCESSED_BY      VARCHAR2(80 BYTE)
);
